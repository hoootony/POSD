#include "Node.h"
#include "ComponentFactory.h"

Node::Node()
{
	_nodeList.clear();
	_type = "Node";
}

Node::Node(string description, int id)
{
	_id = id;
	_description = description;
	_type = "Node";
	_parent = NULL;
}

Node::~Node()
{
}

Component* Node::getParent()

{
